#include <iostream>
using namespace std;

int main() {
	int side;
	scanf("%d", &side);
	int i=0;
	while (i < side) {
		int j = i;
		while (j >= 0) {
			printf("#");
			j--;
		}
		printf("\n");
		i++;
	}
	return 0;
}
